SELECT COUNT(`id`),YEAR(`date_of_birth`)
FROM `ospiti` 
GROUP BY YEAR(`date_of_birth`);

SELECT SUM(`price`),`status`
FROM `pagamenti` 
GROUP BY `status`;

SELECT COUNT(`stanza_id`),`stanza_id`
FROM `prenotazioni` 
GROUP BY `stanza_id`;

SELECT COUNT(HOUR(`created_at`)),HOUR(`created_at`)
FROM `prenotazioni` 
GROUP BY HOUR(`created_at`);
